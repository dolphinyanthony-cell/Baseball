// ============================ // REALISTIC PLAYER MODEL // ============================ function createPlayer(color,x,z,pose){ let player = new THREE.Group(); // BODY let body = new THREE.Mesh( new THREE.CylinderGeometry( .45, .6, 1.5, 16 ), new THREE.MeshStandardMaterial({ color:color }) ); body.position.y=2; player.add(body); // HEAD let head = new THREE.Mesh( new THREE.SphereGeometry( .35, 16, 16 ), new THREE.MeshStandardMaterial({ color:0xffc89b }) ); head.position.y=3.1; player.add(head); // HELMET let helmet = new THREE.Mesh( new THREE.SphereGeometry( .38, 16, 16, 0, Math.PI ), new THREE.MeshStandardMaterial({ color:0x111111 }) ); helmet.position.y=3.25; player.add(helmet); // ARMS let armMaterial = new THREE.MeshStandardMaterial({ color:color }); let arm1 = new THREE.Mesh( new THREE.CylinderGeometry( .12, .12, 1, 12 ), armMaterial ); arm1.position.set( .55, 2.2, 0 ); player.add(arm1); let arm2=arm1.clone(); arm2.position.x=-.55; player.add(arm2); // LEGS let legMaterial = new THREE.MeshStandardMaterial({ color:color }); let leg1 = new THREE.Mesh( new THREE.CylinderGeometry( .15, .18, 1.2, 12 ), legMaterial ); leg1.position.set( .25, 0.8, 0 ); player.add(leg1); let leg2=leg1.clone(); leg2.position.x=-.25; player.add(leg2); player.position.set( x, 0, z ); // pitcher stance if(pose==="pitcher"){ player.rotation.y=Math.PI; } scene.add(player); return player; } // ============================ // BAT MODEL // ============================ function createBat(){ let bat = new THREE.Group(); let barrel = new THREE.Mesh( new THREE.CylinderGeometry( .12, .18, 3, 16 ), new THREE.MeshStandardMaterial({ color:0x9b5a20 }) ); barrel.rotation.z=Math.PI/2; bat.add(barrel); let grip = new THREE.Mesh( new THREE.CylinderGeometry( .07, .07, .7, 16 ), new THREE.MeshStandardMaterial({ color:0x222222 }) ); grip.position.x=-1.7; grip.rotation.z=Math.PI/2; bat.add(grip); return bat; } // ============================ // CREATE PLAYERS // ============================ let batter = createPlayer( 0x0055ff, 0, 8, "batter" ); let pitcher = createPlayer( 0xff2222, 0, -30, "pitcher" ); // ============================ // CREATE BAT // ============================ let bat=createBat(); bat.position.set( 1, 2.4, 7 ); scene.add(bat);
